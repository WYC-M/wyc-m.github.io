    import random

    list=[]

    all=input("总人数 ")
    luck=input("中签人数 ")

    i=0

    for i in range (eval(all)):
        list.append(i+1)
        i=i+1

    random.shuffle(list)
    i=0

    print("中签者",end=" ")

    for i in range (eval(luck)):
        print(list[i],end=" ")
        i=i+1

    input()
